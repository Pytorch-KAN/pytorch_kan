Quick Start Example

import torch
from pytorch_kan import KAN

# Create a simple KAN model
model = KAN(
    input_size=10,
    output_size=1,
    hidden_layers=[32, 32],
    basis_func='chebyshev',
    grid_size=16
)

# Train like any PyTorch model
optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
